# Toast

## Basic Usage

<Canvas>
  <Story name="Basic Usage">
    {() => {
      const [visible, setVisible] = useState(false)
      return (
        <>
          <Toast
            visible={visible}
            onClose={() => {
              setVisible(false)
            }}
          >
            This is a toast.
          </Toast>
          <button
            className="yobta-button-primary"
            type="button"
            onClick={() => {

          </button>
        </>
      )
    }}
  </Story>

## Caching

For performance reasons, the toast is not updating it's children during animation phase. Set `noCache={true}` to force a re-render.

Please notice that the toast content flickers when exiting and not cached. Default setup prevents flcikering.

<Canvas>
  <Story name="Caching">
    {() => {
      const [errors, setErrors] = useState([])
      const [noCache, setNoCache] = useState(false)
      const [, setVisible] = useState(false)
      const visible = errors.length > 0
      const popError = () => setErrors(([, ...nextErrors]) => nextErrors)
      const pushError = newError =>
        setErrors(lastErrors => [...lastErrors, newError])
      return (
        <>
          <Toast visible={visible} noCache={noCache} onClose={popError}>
            {({ countdown, state, close }) => {
              return (
                <>
                  <div>
                    <p>{errors[0]}.</p>
                    <p className="yobta-badge yobta-paper-2-inversed">
                      {state}
                    </p>
                  </div>
                  <button
                    className="yobta-button p-0 rounded-full w-8 h-8"
                    onClick={close}
                  >
                    <Cross />
                  </button>
                </>
              )
            }}
          </Toast>
          <div className="flex gap-x-2">
            <button
              className="yobta-button-primary"
              type="button"
              onClick={() => {
                pushError(`Created: ${Date.now()}`)
              }}
            >
              Toasts: {errors.length}
            </button>
            <button
              className="yobta-button-secondary"
              type="button"
              onClick={() => {
                setNoCache(last => !last)
              }}
            >
              Cache: {noCache ? 'off' : 'on'}
            </button>
          </div>
        </>
      )
    }}
  </Story>
</Canvas>

## Visiblity State

<Canvas>
  <Story name="Visiblity State">
    {() => {
      const [state, setState] = useState('idle')
      const [visible, setVisible] = useState(false)
      return (
        <>
          <Toast
            visible={visible}
            onClose={() => {
              setVisible(false)
            }}
            onEnter={() => {
              setState('entered')
            }}
            onLeave={() => {
              setState('left')
            }}
          >
            See the console
          </Toast>
          <button
            className="yobta-button-primary"
            type="button"
            onClick={() => {
              setVisible(last => !last)
            }}
          >
            Toast: {state}
          </button>
        </>
      )
    }}
  </Story>
</Canvas>

## Offset

<Canvas>
  <Story name="Offset">
    {() => {
      const [offset, setOffset] = useState(8)
      const [visible, setVisible] = useState(false)
      return (
        <>
          <Toast
            onClose={() => {
              setVisible(false)
            }}
            offset={offset}

            className="yobta-button-primary"

            onClick={() => {
              setVisible(last => !last)

          >
            Toast: {offset}
          </button>
          <br />
          <input
            defaultValue={offset}
            min={0}
            max={240}
            onChange={event => {
              setOffset(Number(event.target.value))
            }}
            type="range"
          />
        </>
      )
    }}
  </Story>
</Canvas>
